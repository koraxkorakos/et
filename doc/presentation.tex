\begin{frame}{Evaluation = Projection on AST}

\centering

\scalebox{0.85}{ % <-- adjust (0.75–0.9 works well)
\begin{tikzpicture}[
    node distance=1.4cm,
    every node/.style={draw, circle, minimum size=7mm},
    op/.style={draw, rectangle, minimum width=7mm, minimum height=7mm},
    idx/.style={draw=none}
]

% AST
\node[op] (minus) {$-$};
\node[op, below left=of minus] (plus) {$+$};
\node[below right=of minus] (c) {$c$};

\node[below left=of plus] (a) {$a$};
\node[below right=of plus] (b) {$b$};

\draw[->] (plus) -- (minus);
\draw[->] (c) -- (minus);
\draw[->] (a) -- (plus);
\draw[->] (b) -- (plus);

% Index set
\node[idx, right=3.5cm of minus] (iset) {$I = \{1,4,5,9\}$};
\draw[->, dashed] (iset) -- (minus);

% Evaluation
\node[idx, below=2.8cm of minus] (res) {
$\begin{aligned}
1 &\mapsto eval\_at(1, expr) \\
4 &\mapsto eval\_at(4, expr) \\
5 &\mapsto eval\_at(5, expr) \\
9 &\mapsto eval\_at(9, expr)
\end{aligned}$
};

\draw[->] (minus) -- (res);

\end{tikzpicture}
}

\vspace{0.8em}

\textbf{Key idea:} We do not traverse the tree — we query it.
